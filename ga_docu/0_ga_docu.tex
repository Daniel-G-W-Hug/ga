\newcommand{\pd}[2]{\ensuremath{\frac{\partial {#1}}{\partial {#2}}}}

\newcommand{\grprj}[2]{\ensuremath{\langle {#1} \rangle_{#2}}}

% norms
\newcommand{\nrm}[1]{\left \lVert {#1} \right \rVert}

% bulk and weight (by adding a filled or empty circle as right downstairs index)
\newcommand{\bulk}[1]{\ensuremath{{#1}_\bullet}}
\newcommand{\weight}[1]{\ensuremath{{#1}_\circ}}

% in addition to \nrm{} we need \bulknrm and \weightnrm{}
\newcommand{\bulknrm}[1]{\ensuremath{\bulk{\nrm{#1}}}}
\newcommand{\weightnrm}[1]{\ensuremath{\weight{\nrm{#1}}}}

% dual definitions
\newcommand{\leftbulkdual}[1]{\ensuremath{{#1}_\star}}          %% filled star
\newcommand{\rightbulkdual}[1]{\ensuremath{{#1}^\star}}
\newcommand{\leftweightdual}[1]{\ensuremath{{#1}_\smallstar}}   %% unfilled star
\newcommand{\rightweightdual}[1]{\ensuremath{{#1}^\smallstar}}

% parallel sign scaled to the hight of the perpendicular sign (package scalerel)
\newcommand*{\parall}{\stretchrel*{\parallel}{\perp}}

\newcommand{\revdagger}[1]{\ensuremath{{#1}^\dagger}}

\newcommand{\rev}[1]{\ensuremath{\widetilde{#1}}}
\newcommand{\rrev}[1]{\ensuremath{\undertilde{#1}}}

\newcommand{\lcmpl}[1]{\ensuremath{\underline{#1}}}
